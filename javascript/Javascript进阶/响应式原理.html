# 什么是响应式

// 对象的响应式
const obj = {
     name:"abc",
     age:18
}

console.log('hello word') // 此处代码重新执行
console.log('你好啊,JavaScript') // 此处代码重新执行

obj.name = 'kobe' // 如果obj里面的某个属性发生了改变,那么上面的代码重新执行

# 响应式函数的封装

// 首先封装一个响应式函数
let reactiveFns = []; // obj.name发生改变的所有需要重新执行的函数
function watchFn(fn){
     reactiveFns.push(fn)
}
// 将需要相应的函数传入封装的响应式函数中
watchFn(function(){
     console.log('需要相应的函数')
})

// 将没有一个响应式函数遍历出来并执行
reactiveFns.forEach(fn => fn());

# 依赖收集类的封装

// 利用ES6类来管理响应式函数
class Depend{
     constructor(){
          this.reactiveFns = [];
     }
     addDepend(reactiveFn){
          this.reactiveFns.push(reactiveFn)
     }
     notify(){
          this.reactiveFns.forEach(fn =>{
               fn()
          })
     }
}
const depend = new Depend()
function watchFn(fn){
     depend.addDepend(fn)
}

watchFn(function(){
     console.log('需要相应的函数')
})
depend.notify();

# 自动监听对象变化

// 使用Proxy代理obj对象
const objProxy = new Proxy(obj,{
     get:function(targe,key,receiver){
          return Reflect.get(targe,key,receiver)
     },
     set:function(target,key,newValue,receiver){
          Reflect.set(target,key,newValue,receiver)
          depend.notify() // 当属性发生了改变,自动执行响应式函数

})
